Sisa $p(x)+q(x)$ dibagi $x^2-1$ adalah \ldots
\begin{pilB}
  \item $3x-1$ \item $3x+1$ \item $-3x+2$ \item $-3x-2$ \item $3x+2$
\end{pilB}
\end{soal}
